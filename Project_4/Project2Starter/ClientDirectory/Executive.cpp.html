std::vector<std::string> CodePublisher::publish(int argc, char** argv)   

{
	
	try 
	{
		convertedfiles.clear();
		std::vector<std::string> filePaths;
		std::vector<std::string> fileNames;
		std::vector<std::string> processedFiles;
		Loader ld;
		Conversion conv;
		filePaths = ld.filenameDisplay(argc, argv);
		if (filePaths.empty())
		{
			std::cout << "\nNo files matched the given Regex!\n";
			
		}
		std::string tmp;

		fileNames = ld.onlyfileNamesreturn();
		processedFiles = conv.filetoString(filePaths, fileNames);
		
//Display disp;

		
		for (auto file : fileNames) {
			tmp = FileSystem::Path::getFullFileSpec(".");
			tmp = FileSystem::Path::getFullFileSpec("../ConvertedPages/" + file + ".html");
			tmp = FileSystem::Path::getName(tmp);
			convertedfiles.push_back(tmp);
		}

		return convertedfiles;
		
	}
	catch (...) 
	{
		std::cout << "Error";
	}
}
