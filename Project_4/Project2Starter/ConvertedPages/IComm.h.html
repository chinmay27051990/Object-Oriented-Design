#include <string>

namespace MsgPassingCommunication
{
  class IComm

  {
  public:
    static IComm* create(const std::string& machineAddress, size_t port);
    virtual void start(const std::string& recPath = "../SavedFiles", const std::string& sndPath = "../SendFiles" ) = 0;
    virtual void stop() = 0;

    virtual ~IComm() {}

  };
